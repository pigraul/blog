## schema参数类型不匹配问题

自动生成的`nextflow_schema.json`将`nextflow.config`的`array`识别为`string`格式，
```
// nextflow.config的params部分
valid_assays = ['rna']

// nextflow_schema.json
"valid_assays": {
    "type": "string",
    "default": "['rna']"
},
```

会产生如下报错：
```
* --valid_assays (["rna"]): Value is [array] but should be [string]
```

修正为

单一选项形式
```
"valid_assays": {
    "type": "string",
    "default": "rna",                        // 直接给字符串值，不加方括号
    "enum": ["rna", "bulk_rna", "starsolo", "scrna_atac"],  // 可选值列表
    "description": "Type of assay to process"
}
```
运行时`--valid_assays rna`

多选时
```
"valid_assays": {
    "type": "array",                      // 改为 array
    "default": ["rna"],                   // 默认值用数组格式
    "items": {
        "type": "string",
        "enum": ["rna", "bulk_rna", "starsolo", "scrna_atac"]
    },
    "description": "List of assays to process (can select multiple)"
}
```
运行时`--valid_assays '["rna", "bulk_rna"]'`

## NF 自动编辑文档

***编辑nextflow_schema.json***

使用`nf-core`根据`nextflow.config`来生成`nextflow_schema.json`。
```
nf-core schema build -d <Pipeline directory>
```
创建`nextflow_schema.json`后，可以将其上传至[Parameter schema](https://nf-co.re/pipeline_schema_builder)可视化编辑。
<br>

***编辑parameters.md***

安装`jsonschema-markdown`后根据`nextflow_schema.json`生成`parameters.md`。
```
python3 -m pip install --user pipx
pipx install jsonschema-markdown
jsonschema-markdown nextflow_schema.json > parameters.md
```

## NF中samplesheetToList报错问题

```
Channel
    .fromList(samplesheetToList(params.input, "${projectDir}/assets/schema_input.json"))
    .map {
        meta, fastq_1, fastq_2 ->
            if (!fastq_2) {
                return [ meta.id, meta + [ single_end:true ], [ fastq_1 ] ]
            } else {
                return [ meta.id, meta + [ single_end:false ], [ fastq_1, fastq_2 ] ]
            }
    }
    .groupTuple()
    .map {
        validateInputSamplesheet(it)
    }
    .map {
        meta, fastqs ->
            return [ meta, fastqs.flatten() ]
    }
    .set { ch_samplesheet }
```

在第22行明明已经只提取meta和fastq的信息，但是在解析的时候会出现额外的信息，会有第四个元素[]：
```
ERROR ~ Invalid method invocation `call` with arguments: [[id:data1M, expected_cells:[]], /test_data/dyn_fastq2/DYNOLC250102035_1M_R1.fastq.gz, /test_data/dyn_fastq2/DYNOLC250102035_1M_R2.fastq.gz, []] (java.util.ArrayList) on _closure12 type
```

问题出现在`schema_input.json`文件的定义中，`match_barcode`部分没有定义`meta`信息，指明`meta`的，都会放在第一个元素里，未知名的在第四个空list中。所以除fastq外，都需要指明`meta`
```
    "items": {
        "type": "object",
        "properties": {
            "sample": {
                "type": "string",
                "pattern": "^\\S+$",
                "meta": ["id"]
            },
            "fastq_1": {
                "type": "string",
                "format": "file-path",
                "exists": true,
                "pattern": "^([\\S\\s]*\\/)?[^\\s\\/]+\\.f(ast)?q\\.gz$"
            },
            "fastq_2": {
                "type": "string",
                "format": "file-path",
                "exists": true,
                "pattern": "^([\\S\\s]*\\/)?[^\\s\\/]+\\.f(ast)?q\\.gz$"            },
            "expected_cells": {
                "type": "integer",
                "meta": ["expected_cells"]
            },
            "match_barcode": {
                "type": "string",
                "format": "file-path",
                "exists": true,
                "pattern": "^\\S*barcodes.tsv*"            }
        },
        "required": ["sample", "fastq_1"]
    }
```

```
        "rna/saturation": {
            "fn": "*rna.saturation.json",
        },
        "rna/median_gene": {
            "fn": "*rna.median_gene.json",
        },
    }
```
